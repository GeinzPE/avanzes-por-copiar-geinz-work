@Composable
fun pantalla_geinz_inmobiliaria(nombre_user: String, coneccion: Boolean, localidad_user: String) {
    val viewmodel: viewmodel_inmobiliaria = viewModel()
    val context = LocalContext.current
    val listarepo by viewmodel.estado_carga_inmuebles_principales.collectAsState()
    var mostrar_carga_Respuesta_gemini by remember { mutableStateOf(false) }

    LaunchedEffect(listarepo) { }

    LaunchedEffect(Unit) {
        viewmodel.obtener_inmubles_dados(localidad_user)
    }
    Box(
        modifier = Modifier
            .fillMaxSize()
    ) {
        Column() {
            texto_generico_multilinea(
                "House capital group",
                style = MaterialTheme.typography.banerGeinzWork
            )
            spacer_vertical(5.dp)
            texto_generico_multilinea(
                "Obten lo mejore de barranca y mira los mejores precios que traemos para ti ",
                style = MaterialTheme.typography.bodyMedium
            )
            spacer_vertical(10.dp)

            when (listarepo) {
                is viewmodel_inmobiliaria.estado_carga_principal_immubles.empty -> {

                }

                is viewmodel_inmobiliaria.estado_carga_principal_immubles.error -> {

                }

                viewmodel_inmobiliaria.estado_carga_principal_immubles.idle -> {

                }

                viewmodel_inmobiliaria.estado_carga_principal_immubles.loading -> {
                    ShimmerImagenConMarca("HOUSE CAPITAL GROUP")
                }

                is viewmodel_inmobiliaria.estado_carga_principal_immubles.succes -> {
                    val lista_inmubles =
                        (listarepo as viewmodel_inmobiliaria.estado_carga_principal_immubles.succes).lista_inmuebles

                    LazyColumn() {
                        items(lista_inmubles) { i ->
                            estilo_visual_card(
                                context,
                                i.lista_img,
                                i.nombre_inmobiliara,
                                i.descripcion,
                                "https://firebasestorage.googleapis.com/v0/b/geinzworkapp.appspot.com/o/logo_geinz_webp.webp?alt=media&token=aa1ef1df-1bcd-48f2-9cad-a85929c3a8d0"
                            ) {

                            }
                        }
                    }

                }
            }
        }
    }

}


@Composable
fun estilo_visual_card(
    context: Context,
    lista_img: List<String>,
    nombre: String,
    descripcion: String,
    logo_capital: String,
    img_clikeble: () -> Unit
) {
    Box(
        modifier = Modifier
            .fillMaxWidth()
            .clip(RoundedCornerShape(bottomStart = 20.dp, bottomEnd = 20.dp))

    ) {
        GaleriaHorizontalInstagram(
            imagenes = lista_img,
            modifier = Modifier.fillMaxSize(), img_clikeble_valor = { select ->
                img_clikeble()
            }, long_listatener = {
                Log.d("LONG_PRESS", "Long press en la galería")
            })
    }
    Row(
        modifier = Modifier
            .padding(start = 4.dp, end = 10.dp, top = 8.dp, bottom = 8.dp)
            .fillMaxWidth(),
        verticalAlignment = Alignment.CenterVertically
    ) {

        // Logo de la tienda
        AsyncImage(
            model = ImageRequest.Builder(LocalContext.current)
                .data(logo_capital)
                .placeholder(R.drawable.cargando_img_categorias)
                .error(R.drawable.cargando_img_categorias)
                .build(),
            contentDescription = null,
            modifier = Modifier
                .size(45.dp)
                .clip(CircleShape)
                .clickable(
                    indication = null,
                    interactionSource = remember { MutableInteractionSource() }
                ) {},
            contentScale = ContentScale.Crop
        )

        Spacer(modifier = Modifier.width(8.dp))


        Column(modifier = Modifier.weight(1f)) {
            Text(
                text = nombre.capitalizeFirst(),
                style = MaterialTheme.typography.bodyMedium,
                maxLines = 1,
                overflow = TextOverflow.Ellipsis
            )

//            Spacer(modifier = Modifier.height(2.dp))
//
//            if (i.informacion_publcacion.titulo.isNotEmpty()) {
            Text(
                text = descripcion.capitalizeFirst(),
                style = MaterialTheme.typography.bodySmall,
                maxLines = 1,
                overflow = TextOverflow.Ellipsis
            )
//            }


        }

        spacer_horizonta(10.dp)
        Row(
            horizontalArrangement = Arrangement.spacedBy(10.dp),
            verticalAlignment = Alignment.CenterVertically
        ) {


            Icon(
                painterResource(R.drawable.whatsapp_icon),
                contentDescription = "WhatsApp",
                modifier = Modifier
                    .size(30.dp)
                    .clickable {},
                tint = Color.Unspecified
            )
        }

    }

}
